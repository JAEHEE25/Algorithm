# [G5] AC

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/5430
> 
> **난이도**: 4
>
> **유형**: #자료구조


### [ 문제 풀이 ]
```Java
import java.util.*;
import java.lang.*;
import java.io.*;

class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int T = Integer.parseInt(br.readLine());
        StringBuilder answer = new StringBuilder();
        
        for (int t = 0; t < T; t++) {
            Deque<String> deque = new LinkedList<>();
            
            //입력 받기
            String func = br.readLine();
            int n = Integer.parseInt(br.readLine());
            String input = br.readLine();
            String[] arr = input.substring(1, input.length() - 1).split(",");
            
            for (String a : arr) {
                if (!a.equals("")) deque.offer(a);
            }
            
            boolean reverse = false;
            for (int i = 0; i < func.length(); i++) { //함수
                char p = func.charAt(i);
                
                if (p == 'R') {
                    reverse = !reverse;
                    continue;
                }

                if (deque.isEmpty()) {
                    break;
                }

                if (reverse) { //뒤집어졌을 경우 마지막 원소 빼기
                    deque.pollLast();
                } else { //원래 방향일 경우 첫번째 원소 빼기
                    deque.pollFirst();
                }
            }

            
            if (deque.isEmpty()) {
                answer.append("error");
            } else {
                StringBuilder sb = new StringBuilder();
                sb.append("[");
                
                if (reverse) {
                    while (true) {
                        sb.append(deque.pollLast());
                        if (deque.isEmpty()) break;
                        sb.append(",");
                    }
                } else {
                    while (true) {
                        sb.append(deque.pollFirst());
                        if (deque.isEmpty()) break;
                        sb.append(",");
                    }                    
                }
                sb.append("]");
                answer.append(sb);
            }
            answer.append("\n");
        }
        
        System.out.println(answer);
    }
}
```
